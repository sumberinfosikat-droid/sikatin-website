        // ============ PREVIEW ============
        function generatePreviewHTML() {
            let html = '';
            blocks.forEach(b => {
                const div = document.createElement('div');
                div.innerHTML = b.content || '';
                const text = div.innerHTML;
                switch(b.type) {
                    case 'heading-2': html += `
${text}
`; break;
                    case 'heading-3': html += `
${text}
`; break;
                    case 'heading-4': html += `
${text}
`; break;
                    case 'paragraph': if(text.trim()) html += `
${text}
`; break;
                    case 'quote': html += `
${text}
`; break;
                    case 'list-ul': case 'list-ol': {
                        const tag = b.type === 'list-ul' ? 'ul' : 'ol';
                        const items = div.textContent.split('\n').filter(Boolean);
                        html += `<${tag}>${items.map(i => `
  • ${i.trim()}
    • `).join('')}`;
                        break;
                    }
                    case 'image':
                        if (b.content) {
                            html += ``;
                            if (b.caption) html += ``;
                        }
                        break;
                    case 'divider': html += '
    '; break;
                    case 'ad-slot': html += '
    Google Ads Slot
    '; break;
                    case 'code': html += `
    ${escapeHTML(div.textContent)}
    `; break;
                }
            });
            return html;
        }
